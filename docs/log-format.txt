Log file (CSV) entry formats (Version of 23-Apr-14) -CR

<DATETIME>,...

* <DATETIME> in format automatically recognized by Excel, e.g.,

  "4/23/14 09:15:33",...

-------------------------------------------------------

<DATETIME>,ID,<CONDITION>,<MACHINE>,<INSTALLED>,<BOOTED>

* <CONDITION> = ALWAYS or LOGIN or REETI
* <MACHINE> = "..."
* <INSTALLED> = <DATETIME>
* <BOOTED> = <DATETIME>

-------------------------------------------------------

<DATETIME>,ENGAGEMENT,<ENGAGEMENT_STATE>,<OLD_STATE>,<NEW_STATE>
      
* <OLD_STATE>/<NEW_STATE> = IDLE or ATTENTION or INITIATION or ENGAGED or RECOVERING

-------------------------------------------------------

<DATETIME>,SESSION,START,<CLOSENESS>,<TIME_OF_DAY>

* <CLOSENESS> = STRANGER or ACQUAINTANCE or COMPANION
* <TIME_OF_DAY> = MORNING or AFTERNOON or EVENING or NIGHT

<DATETIME>,SESSION,END,<DISENGAGEMENT>,<DURATION>,<REPEATS>

* <DISENGAGEMENT> = GOODBYE or TIMEOUT
* <DURATION> in minutes
* <REPEATS> number of times user clicked "What did you say?"

-----------------------------------------------------------

<DATETIME>,<ACTIVITY>,PROPOSED
<DATETIME>,<ACTIVITY>,ACCEPTED
<DATETIME>,<ACTIVITY>,REJECTED
<DATETIME>,<ACTIVITY>,START
<DATETIME>,<ACTIVITY>,END

* <ACTIVITY> = SESSION or
             ABOUT or
             ANECDOTES or
             CALENDAR or 
             CHECKERS or 
             ENROLL or
             EXERCISE or
             EXPLAIN or
             GOODBYE or
             GREETINGS or 
             HEALTH or
             NUTRITION or
             RUMMY or 
             SKYPE or
             STORY or 
             TTT or
             WEATHER

<DATETIME>,<ACTIVITY>,INTERRUPTION,CALENDAR,<ENTRY>

* <ENTRY> = "..." (via toString method of CalendarEntry)

<DATETIME>,<ACTIVITY>,INTERRUPTION,SKYPE,<CALLER>

* <CALLER> = "..."

<DATETIME>,<ACTIVITY>,SAY,<UTTERANCE>

* <UTTERANCE> = "..."

<DATETIME>,<ACTIVITY>,MENU,<CHOICES>...
<DATETIME>,<ACTIVITY>,EXTENSION,<CHOICES>...

* <CHOICES> = "...",...

<DATETIME>,<ACTIVITY>,SELECTED,<SELECTED>

* <SELECTED> = "..."

<DATETIME>,ACTIVITY,<PLUGIN>,KEYBOARD,<INPUT>

* <INPUT> = "..."
---------------------------------------------------



<DATETIME>,ACTIVITY,<PLUGIN>,USER,<PROPERTY>,<VALUE>

* <PROPERTY> = "..."
* <VALUE> = "..."

---------------------------------------------------------

<DATETIME>,ACTIVITY,ABOUT,<POSITIVE>,<NEGATIVE>,<EXIT>

* <POSITIVE> = POSITIVE or NOT_POSITIVE
* <NEGATIVE> = NEGATIVE or NOT_NEGATIVE
* <EXIT> = EARLY or NORMAL

---------------------------------------------------------

<DATETIME>,ACTIVITY,ANECDOTES,<FINISHED>,<TITLE>

* <FINISHED> = FINISHED or NOT_FINISHED
* <TITLE> = "..."


---------------------------------------------------------

<DATETIME>,ACTIVITY,CALENDAR,UPDATE,<STATUS>,<ENTRY>

* <STATUS> = ADD or CHANGE or DELETE

<DATETIME>,ACTIVITY,CALENDAR,VIEW,<MODE>,<DATETIME>
        
* <MODE> = DAY or WEEK or MONTH

---------------------------------------------------------

<DATETIME>,ACTIVITY,CHECKERS,<WON>,<JUMPS>,<DOUBLES>

* <WON> = USER or AGENT or NEITHER
* <JUMPS> = total number of jumps in game
* <DOUBLES> = total number of double jumps in game

---------------------------------------------------------

<DATETIME>,ACTIVITY,ENROLL,<CHANGE>,<NAME>

* <CHANGE> = NEW or EDIT
* <NAME> = "..."  (see details of person in user model printout)

---------------------------------------------------------

<DATETIME>,ACTIVITY,EXERCISE,<TOPIC>

* <TOPIC> = GOALS or ACTIVITIES or SERVINGS

---------------------------------------------------------

<DATETIME>,ACTIVITY,EXPLAIN,<PLUGIN>

---------------------------------------------------------

<DATETIME>,ACTIVITY,GREETINGS,<TOD>

* <TOD> = MORNING or AFTERNOON or EVENING or NIGHT

---------------------------------------------------------

<DATETIME>,ACTIVITY,SRUMMY,<WON>,<FIRST>,<USER_MELDS>,<AGENT_MELDS>

* <FIRST> = USER or AGENT
* <USER_MELDS> = number of user melds in game
* <AGENT_MELDS> = number of agent melds in game

---------------------------------------------------------

<DATETIME>,ACTIVITY,SKYPE,<DIRECTION>,<APPOINTMENT>,<DURATION>,<WHO>

* <DIRECTION> = INCOMING or OUTGOING
* <APPOINTMENT> = APPOINTMENT or NOT_APPOINTMENT
* <WHO> = "..." (Skype name)

---------------------------------------------------------

<DATETIME>,ACTIVITY,STORY,<SAVED>,<DURATION>,<TITLE>

* <SAVED> = SAVED or NOT_SAVED
* <TITLE> = "..."
---------------------------------------------------------

<DATETIME>,ACTIVITY,TTT,<WON>,<FIRST>

---------------------------------------------------------

<DATETIME>,ACTIVITY,WEATHER,<OPTION>

* <OPTION> = CITIES or FRIENDS
