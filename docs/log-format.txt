Log file (CSV) entry formats (Version of 23-Apr-14) -CR

<TIMESTAMP>...

* <DATESTAMP> in format automatically recognized by Excel, e.g.,

  "4/23/14 09:15:33",...

* <TYPE> = SESSION or ACTIVITY

-------------------------------------------------------

<TIMESTAMP>,SESSION,ATTEMPTED,<STATE>

* <STATE> = ATTENTION or INITIATING
            (state reached before going back to idle)
            (there will be no such entries for login condition)

<TIMESTAMP>,SESSION,START
<TIMESTAMP>,SESSION,END,<DURATION>,<REPEATS>,<DISENGAGMENT>

* <DURATION> in minutes
* <REPEATS> number of times user clicked "What did you say?"
* <DISENGAGEMENT> = GOODBYE or TIMEOUT

<TIMESTAMP>,SESSION,CLOSENESS,<LEVEL>

* <LEVEL> = STRANGER or ACQUAINTANCE or COMPANION

<TIMESTAMP>,SESSION,REMINDER,<EVENT>

* <EVENT> = "..."

<TIMESTAMP>,SESSION,AGENT,<UTTERANCE>
<TIMESTAMP>,SESSION,MENU,<CHOSEN>,<CHOICES>...

* <UTTERANCE> = "..."

* <CHOSEN> = "..."

* <CHOICES> = "...",...

---------------------------------------------------

<TIMESTAMP>,ACTIVITY,<PLUGIN>,...

* <PLUGIN> = ABOUT or
             ANECDOTES or
             CALENDAR or 
             CHECKERS or 
             ENROLL or
             EXERCISE or
             EXPLAIN or
             GREETINGS or 
             HEALTH or
             NUTRITION or
             SRUMMY or 
             SKYPE or
             STORY or 
             TTT or
             WEATHER

<TIMESTAMP>,ACTIVITY,<PLUGIN>,PROPOSED,<ACCEPTED>

* <ACCEPTED> = ACCEPTED or NOT_ACCEPTED

<TIMESTAMP>,ACTIVITY,<PLUGIN>,START
<TIMESTAMP>,ACTIVITY,<PLUGIN>,END,<DURATION>,<REPEATS>

<TIMESTAMP>,ACTIVITY,<PLUGIN>,AGENT,<UTTERANCE>
<TIMESTAMP>,ACTIVITY,<PLUGIN>,MENU,<CHOSEN>,<CHOICES>...

<TIMESTAMP>,ACTIVITY,<PLUGIN>,KEYBOARD,<INPUT>

* <INPUT> = "..."

<TIMESTAMP>,ACTIVITY,<PLUGIN>,USER,<PROPERTY>,<VALUE>

* <PROPERTY> = "..."
* <VALUE> = "..."

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,ABOUT,<POSITIVE>,<NEGATIVE>,<EXIT>

* <POSITIVE> = POSITIVE or NOT_POSITIVE
* <NEGATIVE> = NEGATIVE or NOT_NEGATIVE
* <EXIT> = EARLY or NORMAL

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,ANECDOTES,<FINISHED>,<TITLE>

* <FINISHED> = FINISHED or NOT_FINISHED
* <TITLE> = "..."


---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,CALENDAR,EVENT,<STATUS><TITLE>

* <STATUS> = ADD or CHANGE or DELETE
* <TITLE> = "..." (see details of event in user model printout)

<TIMESTAMP>,ACTIVITY,CALENDAR,VIEW,<MODE>,<DATE>

* <MODE> = DAY or WEEK or MONTH
* <DATE> = "..."

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,CHECKERS,<WON>,<JUMPS>,<DOUBLES>

* <WON> = USER or AGENT or NEITHER
* <JUMPS> = total number of jumps in game
* <DOUBLES> = total number of double jumps in game

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,ENROLL,<CHANGE>,<NAME>

* <CHANGE> = NEW or EDIT
* <NAME> = "..."  (see details of person in user model printout)

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,EXERCISE,<TOPIC>

* <TOPIC> = GOALS or ACTIVITIES or SERVINGS

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,EXPLAIN,<PLUGIN>

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,GREETINGS,<TOD>

* <TOD> = MORNING or AFTERNOON or EVENING or NIGHT

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,HEALTH,...

* TBD

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,NUTRITION,...

* TBD

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,SRUMMY,<WON>,<FIRST>,<USER_MELDS>,<AGENT_MELDS>

* <FIRST> = USER or AGENT
* <USER_MELDS> = number of user melds in game
* <AGENT_MELDS> = number of agent melds in game

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,SKYPE,<DIRECTION>,<APPOINTMENT>,<DURATION>,<WHO>

* <DIRECTION> = INCOMING or OUTGOING
* <APPOINTMENT> = APPOINTMENT or NOT_APPOINTMENT
* <WHO> = "..." (Skype name)

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,STORY,<SAVED>,<DURATION>,<TITLE>

* <SAVED> = SAVED or NOT_SAVED
* <TITLE> = "..."
---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,TTT,<WON>,<FIRST>

---------------------------------------------------------

<TIMESTAMP>,ACTIVITY,WEATHER,<OPTION>

* <OPTION> = CITIES or FRIENDS
